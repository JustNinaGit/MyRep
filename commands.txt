git init - створення репозиторію

git status - перевірка стану репозиторію

git add - додавання файлів до індексу

git commit - створення коміту

git log - перегляд історії комітів

git branch - перегляд гілок

git checkout - перехід між гілками

git merge - об'єднання гілок

git push - відправка змін на GitHub

git pull - отримання змін з GitHub
